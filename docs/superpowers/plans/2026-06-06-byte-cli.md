> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Go CLI tool `sancho` that delegates coding tasks to LLMs via OpenRouter, AWS Bedrock, OpenAI, and Anthropic, with `ask` for analysis and `write` for code generation.

**Architecture:** Cobra CLI with layered config resolution (CLI > file > env > defaults), provider abstraction via interface with retry/timeout middleware, and minimal output modes (MinLang for ask, plain text for write).

**Tech Stack:** Go, Cobra, AWS SDK v2 (bedrock only), standard library HTTP + JSON

## Logging & Error Handling

All code in this project uses a single error and logging strategy defined here and followed by every task below.

### Error types (`internal/errors/errors.go`)

Three typed errors propagate through the entire app. No other error types exist in internal packages.

- `UserError` — expected failure (bad input, missing file, auth). The cmd layer prints as `ERROR: <reason>\\n` to stderr.
- `RetryableError` — transient failure (429/502/503/504). Carries `RetryAfter time.Duration` for backoff. Retry middleware checks `errors.As(err, &errors.RetryableError{})`.
- `InternalError` — unexpected failure. The cmd layer prints as `INTERNAL: <reason>\\n`.

```go
package errors

type UserError struct{ Msg string }
type RetryableError struct{ Msg string; RetryAfter time.Duration }
type InternalError struct{ Msg string }

func (e *UserError) Error() string          { return e.Msg }
func (e *RetryableError) Error() string     { return e.Msg }
func (e *InternalError) Error() string      { return e.Msg }
```

### Logger (`internal/log/log.go`)

Initialized once in `cmd/root.go`. Every internal package imports `internal/log` and calls `log.Debug`, `log.Info`, `log.Error`.

- Format: JSON to stderr, always. Keys: `ts` (RFC3339), `level`, `msg`.
- Control: no `fmt.Println` or `log.Fatal` in internal packages. The cmd layer prints user-facing messages.
- No `os.Exit` anywhere except `main.go` calling `cmd.Execute()`.

```go
package log

import (
    "encoding/json"
    "log"
    "os"
    "time"
)

var debugEnabled bool
var logger *log.Logger

func Init(debug bool) { ... }
func Debug(msg string, kv ...any) { ... }
func Info(msg string, kv ...any)  { ... }
func Error(msg string, kv ...any) { ... }
```

(Full implementation follows the pattern: marshal entry with timestamp, level, msg, write via `logger.Println`.)

### Propagation rules (followed by every task)

1. Internal packages return `error` (often typed). They never print.
2. `cmd` packages use `cmd.Fatal(err)` which maps `*errors.UserError` → `ERROR: …`, `*errors.InternalError` → `INTERNAL: …`, else `ERROR: <unexpected>`.
3. Retry middleware uses `errors.As` to detect retryable errors and honour `RetryAfter`.
4. Cost footer uses `log.Info` with structured fields: `prompt_tokens`, `completion_tokens`, `total_tokens`, `model`. This is written to stderr even in `--json` mode (human stdout stays clean).

```go
package config

import (
    "encoding/json"
    "os"
    "path/filepath"
    "regexp"
)

func LoadFile(dir string) (Config, error) {
    home := os.Getenv("HOME")
    if home == "" {
        home = os.Getenv("USERPROFILE")
    }

    candidates := []string{
        filepath.Join(dir, ".sancho.json"),
        filepath.Join(home, ".config", "sancho", "config.json"),
    }

    for _, path := range candidates {
        data, err := os.ReadFile(path)
        if err != nil {
            continue
        }
        cleaned := stripJSONC(string(data))
        var cfg Config
        if err := json.Unmarshal([]byte(cleaned), &cfg); err != nil {
            return Config{}, err
        }
        return cfg, nil
    }

    return Config{}, nil
}

var lineComment = regexp.MustCompile(`(?m)//[^\n]*`)
var blockComment = regexp.MustCompile(`(?s)/\*.*?\*/`)

func stripJSONC(s string) string {
    s = lineComment.ReplaceAllString(s, "")
    s = blockComment.ReplaceAllString(s, "")
    return s
}
```

```go
package files

import (
    "bufio"
    "fmt"
    "os"
    "path/filepath"
    "strings"
)

type FileResult struct {
    Path string
    Content string
    Error error
}

func ReadFiles(pattern string) ([]FileResult, error) {
    matches, err := filepath.Glob(pattern)
    if err != nil {
        return nil, err
    }

    if strings.Contains(pattern, "**") {
        dir := filepath.Dir(pattern)
        var recursive []string
        filepath.Walk(dir, func(path string, info os.FileInfo, err error) error {
            if err != nil {
                return nil
            }
            if !info.IsDir() && strings.HasSuffix(path, filepath.Ext(pattern)) {
                rel, _ := filepath.Rel(dir, path)
                if !strings.Contains(rel, "**") {
                    recursive = append(recursive, path)
                }
            }
            return nil
        })
        matches = append(matches, recursive...)
    }

    if len(matches) == 0 {
        return nil, fmt.Errorf("no files match pattern: %s", pattern)
    }

    seen := make(map[string]bool)
    var results []FileResult
    for _, m := range matches {
        if seen[m] {
            continue
        }
        seen[m] = true
        data, err := os.ReadFile(m)
        r := FileResult{Path: m}
        if err != nil {
            r.Error = err
        } else {
            r.Content = string(data)
        }
        results = append(results, r)
    }
    return results, nil
}

func FormatForPrompt(results []FileResult) string {
    var sb strings.Builder
    scanner := bufio.NewScanner(strings.NewReader(""))
    for _, r := range results {
        if r.Error != nil {
            continue
        }
        sb.WriteString(fmt.Sprintf("<file path=\"%s\">%s</file>", r.Path, r.Content))
    }
    return sb.String()
}
```

```go
package client

import (
    "context"
    "math"
    "net/http"
    "time"

    "github.com/marioaer/sancho/internal/errors"
)

type retryProvider struct {
    next        Provider
    maxAttempts int
    backoffMax  time.Duration
}

func WithRetry(p Provider, retry config.RetryConfig) Provider {
    return &retryProvider{
        next:        p,
        maxAttempts: retry.MaxAttempts,
        backoffMax:  time.Duration(retry.BackoffMaxSeconds) * time.Second,
    }
}

func (r *retryProvider) ChatCompletion(ctx context.Context, req ChatRequest) (*ChatResponse, error) {
    var lastErr error
    for attempt := 0; attempt < r.maxAttempts; attempt++ {
        resp, err := r.next.ChatCompletion(ctx, req)
        if err == nil {
            return resp, nil
        }
        lastErr = err
        var retryErr *errors.RetryableError
        if !errors.As(lastErr, &retryErr) {
            return nil, lastErr
        }
        delay := time.Duration(math.Pow(2, float64(attempt))) * time.Second
        if delay > r.backoffMax {
            delay = r.backoffMax
        }
        if retryErr.RetryAfter > 0 && retryErr.RetryAfter < delay {
            delay = retryErr.RetryAfter
        }
        select {
        case <-time.After(delay):
        case <-ctx.Done():
            return nil, ctx.Err()
        }
    }
    return nil, lastErr
}

func (r *retryProvider) SupportsModel(model string) bool {
    return r.next.SupportsModel(model)
}
```

```go
func TestFilePrecedence(t *testing.T) {
    dir := t.TempDir()
    project := filepath.Join(dir, "project")
    os.MkdirAll(project, 0755)
    os.WriteFile(project+"/.sancho.json", []byte(`{"provider":"openrouter"}`), 0644)

    homeDir := filepath.Join(dir, "home", ".config", "sancho")
    os.MkdirAll(homeDir, 0755)
    os.WriteFile(homeDir+"/config.json", []byte(`{"provider":"anthropic"}`), 0644)

    cfg1, _ := LoadFile(project)
    if cfg1.Provider != "openrouter" {
        t.Fatalf("expected project config first, got %s", cfg1.Provider)
    }

    os.Remove(project + "/.sancho.json")
    cfg2, _ := LoadFile(project)
    if cfg2.Provider != "anthropic" {
        t.Fatalf("expected home config fallback, got %s", cfg2.Provider)
    }
}
```
